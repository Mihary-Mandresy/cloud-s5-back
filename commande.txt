# 1. Créer le répertoire du projet
mkdir mon-projet-laravel && cd mon-projet-laravel

# 2. Créer docker-compose.yml
echo 'version: "3.8"

services:
  app:
    image: php:8.2-apache
    container_name: laravel_app
    working_dir: /var/www/html
    volumes:
      - ./:/var/www/html
    ports:
      - "8000:80"
    depends_on:
      - postgres
    environment:
      DB_CONNECTION: pgsql
      DB_HOST: postgres
      DB_PORT: 5432
      DB_DATABASE: laravel_db
      DB_USERNAME: laravel_user
      DB_PASSWORD: secret

  postgres:
    image: postgres:14
    container_name: laravel_postgres
    environment:
      POSTGRES_DB: laravel_db
      POSTGRES_USER: laravel_user
      POSTGRES_PASSWORD: secret
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data

volumes:
  postgres_data:' > docker-compose.yml

# 3. Démarrer les conteneurs
docker-compose up -d

# 4. Entrer dans le conteneur PHP
docker exec -it laravel_app_api bash

# 5. Dans le conteneur, installer les dépendances
apt-get update && apt-get install -y \
    libpq-dev \
    zip \
    unzip \
    git \
    libonig-dev \
    libxml2-dev \
    && docker-php-ext-install pdo pdo_pgsql mbstring xml

# 6. Installer Composer
curl -sS https://getcomposer.org/installer | php -- --install-dir=/usr/local/bin --filename=composer

# 7. Créer le projet Laravel
composer create-project laravel/laravel .

# 8. Configurer la base de données dans .env
sed -i 's/DB_CONNECTION=mysql/DB_CONNECTION=pgsql/' .env
sed -i 's/DB_HOST=127.0.0.1/DB_HOST=postgres/' .env
sed -i 's/DB_PORT=3306/DB_PORT=5432/' .env
sed -i 's/DB_DATABASE=laravel/DB_DATABASE=laravel_db/' .env
sed -i 's/DB_USERNAME=root/DB_USERNAME=laravel_user/' .env
sed -i 's/DB_PASSWORD=/DB_PASSWORD=secret/' .env

# 9. Générer la clé d'application
php artisan key:generate

# 10. Lancer les migrations (depuis le conteneur)
php artisan migrate

# 11. Quitter le conteneur
exit
